type RequestFn = <T = unknown>(method: string, params?: unknown) => Promise<T>;

<action>
Create `src/api/cron.test.ts` with 2-3 tests covering:
1. `list()` — calls `cron.list` with params, returns result
2. `add()` — calls `cron.add` with params, returns result
3. Error propagation — when mock throws, error propagates

Follow the exact pattern from `chat.test.ts`:
```ts
import { describe, it, expect, vi, beforeEach } from 'vitest';
import { CronAPI } from './cron';

type RequestFn = <T = unknown>(method: string, params?: unknown) => Promise<T>;

describe('CronAPI', () => {
  let api: CronAPI;
  let mockRequest: ReturnType<typeof vi.fn>;

  beforeEach(() => {
    mockRequest = vi.fn() as unknown as RequestFn;
    api = new CronAPI(mockRequest as RequestFn);
  });

  describe('list', () => {
    it('should call cron.list with no params', async () => {
      const result = { jobs: [] };
      mockRequest.mockResolvedValueOnce(result);
      const response = await api.list();
      expect(mockRequest).toHaveBeenCalledWith('cron.list', undefined);
      expect(response).toEqual(result);
    });

    it('should call cron.list with params', async () => {
      const params = { limit: 10 };
      const result = { jobs: [{ id: 'job-1' }] };
      mockRequest.mockResolvedValueOnce(result);
      const response = await api.list(params);
      expect(mockRequest).toHaveBeenCalledWith('cron.list', params);
      expect(response).toEqual(result);
    });
  });

  describe('add', () => {
    it('should call cron.add with cron and prompt params', async () => {
      const params = { cron: '0 * * * *', prompt: 'Check status' };
      const result = { jobId: 'job-new' };
      mockRequest.mockResolvedValueOnce(result);
      const response = await api.add(params);
      expect(mockRequest).toHaveBeenCalledWith('cron.add', params);
      expect(response).toEqual(result);
    });
  });
});
```

Also add a test for error propagation and at least one more method (e.g., `remove` or `status`).
